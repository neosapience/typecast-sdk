# Typecast Zig SDK Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a pure Zig SDK for the Typecast TTS API with all endpoints, tests, and documentation.

**Architecture:** Pure Zig using `std.http.Client` (low-level request API) and `std.json`. Callback-based streaming. Zig 0.15+ (uses `Client.request()` + `Request.receiveHead()` + `Reader.bodyReader()` flow).

**Tech Stack:** Zig 0.15.2, std.http.Client, std.json, std.http.Server (mock tests)

```zig
// typecast-zig/build.zig
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    // Library module
    const lib_mod = b.addModule("typecast", .{
        .root_source_file = b.path("src/root.zig"),
        .target = target,
        .optimize = optimize,
    });

    // Unit tests
    const unit_tests = b.addTest(.{
        .root_source_file = b.path("tests/unit_test.zig"),
        .target = target,
        .optimize = optimize,
    });
    unit_tests.root_module.addImport("typecast", lib_mod);

    const run_unit_tests = b.addRunArtifact(unit_tests);
    const test_step = b.step("test", "Run unit tests");
    test_step.dependOn(&run_unit_tests.step);

    // Mock tests
    const mock_tests = b.addTest(.{
        .root_source_file = b.path("tests/mock_test.zig"),
        .target = target,
        .optimize = optimize,
    });
    mock_tests.root_module.addImport("typecast", lib_mod);

    const run_mock_tests = b.addRunArtifact(mock_tests);
    const mock_step = b.step("test-mock", "Run mock server tests");
    mock_step.dependOn(&run_mock_tests.step);

    // Integration tests
    const integration = b.option(bool, "integration", "Run integration tests") orelse false;
    if (integration) {
        const int_tests = b.addTest(.{
            .root_source_file = b.path("tests/integration_test.zig"),
            .target = target,
            .optimize = optimize,
        });
        int_tests.root_module.addImport("typecast", lib_mod);

        const run_int_tests = b.addRunArtifact(int_tests);
        const int_step = b.step("test-integration", "Run integration tests");
        int_step.dependOn(&run_int_tests.step);
    }

    // Example
    const example = b.addExecutable(.{
        .name = "tts_example",
        .root_source_file = b.path("examples/tts.zig"),
        .target = target,
        .optimize = optimize,
    });
    example.root_module.addImport("typecast", lib_mod);

    const run_example = b.addRunArtifact(example);
    const example_step = b.step("example", "Run TTS example");
    example_step.dependOn(&run_example.step);
}
```

```zig
// typecast-zig/src/models.zig
const std = @import("std");

pub const TtsModel = enum {
    ssfm_v21,
    ssfm_v30,

    pub fn toString(self: TtsModel) []const u8 {
        return switch (self) {
            .ssfm_v21 => "ssfm-v21",
            .ssfm_v30 => "ssfm-v30",
        };
    }

    pub fn fromString(s: []const u8) ?TtsModel {
        if (std.mem.eql(u8, s, "ssfm-v21")) return .ssfm_v21;
        if (std.mem.eql(u8, s, "ssfm-v30")) return .ssfm_v30;
        return null;
    }
};

pub const AudioFormat = enum {
    wav,
    mp3,

    pub fn toString(self: AudioFormat) []const u8 {
        return switch (self) {
            .wav => "wav",
            .mp3 => "mp3",
        };
    }
};

pub const EmotionPreset = enum {
    normal,
    happy,
    sad,
    angry,
    whisper,
    toneup,
    tonedown,

    pub fn toString(self: EmotionPreset) []const u8 {
        return switch (self) {
            .normal => "normal",
            .happy => "happy",
            .sad => "sad",
            .angry => "angry",
            .whisper => "whisper",
            .toneup => "toneup",
            .tonedown => "tonedown",
        };
    }
};

pub const Gender = enum {
    male,
    female,

    pub fn toString(self: Gender) []const u8 {
        return switch (self) {
            .male => "male",
            .female => "female",
        };
    }
};

pub const Age = enum {
    child,
    teenager,
    young_adult,
    middle_age,
    elder,

    pub fn toString(self: Age) []const u8 {
        return switch (self) {
            .child => "child",
            .teenager => "teenager",
            .young_adult => "young_adult",
            .middle_age => "middle_age",
            .elder => "elder",
        };
    }
};

pub const SmartPrompt = struct {
    previous_text: ?[]const u8 = null,
    next_text: ?[]const u8 = null,
};

pub const PresetPrompt = struct {
    emotion_preset: EmotionPreset = .normal,
    emotion_intensity: ?f64 = 1.0,
};

pub const Prompt = struct {
    emotion_preset: EmotionPreset = .normal,
    emotion_intensity: ?f64 = 1.0,
};

pub const TtsPrompt = union(enum) {
    smart: SmartPrompt,
    preset: PresetPrompt,
    basic: Prompt,
};

pub const Output = struct {
    volume: ?i32 = 100,
    target_lufs: ?f64 = null,
    audio_pitch: ?i32 = 0,
    audio_tempo: ?f64 = 1.0,
    audio_format: AudioFormat = .wav,
};

pub const OutputStream = struct {
    audio_pitch: ?i32 = 0,
    audio_tempo: ?f64 = 1.0,
    audio_format: AudioFormat = .wav,
};

pub const TtsRequest = struct {
    voice_id: []const u8,
    text: []const u8,
    model: TtsModel,
    language: ?[]const u8 = null,
    prompt: ?TtsPrompt = null,
    output: ?Output = null,
    seed: ?i32 = null,
};

pub const TtsRequestStream = struct {
    voice_id: []const u8,
    text: []const u8,
    model: TtsModel,
    language: ?[]const u8 = null,
    prompt: ?TtsPrompt = null,
    output: ?OutputStream = null,
    seed: ?i32 = null,
};

pub const TtsResponse = struct {
    audio_data: []const u8,
    duration: f64,
    format: AudioFormat,
};

pub const SubscriptionResponse = struct {

    pub const Credits = struct {
```

```zig
// Key structure:
pub const Client = struct {
    allocator: std.mem.Allocator,
    http_client: std.http.Client,
    api_key: []const u8,
    base_url: []const u8,

    pub const Config = struct {
        api_key: []const u8,
        base_url: []const u8 = "https://api.typecast.ai",
    };

    pub fn init(allocator: std.mem.Allocator, config: Config) Client { ... }
    pub fn deinit(self: *Client) void { ... }

    // Private helpers:
    fn doPost(self: *Client, path: []const u8, body: []const u8) !HttpResponse { ... }
    fn doGet(self: *Client, path: []const u8) !HttpResponse { ... }
    fn doPostStream(self: *Client, path: []const u8, body: []const u8, on_chunk: *const fn([]const u8) anyerror!void) !void { ... }
    fn mapHttpError(status: std.http.Status) TypecastError { ... }
};
```

HTTP flow for Zig 0.15:
1. `client.http_client.request(.POST, uri, .{ .extra_headers = ... })`
2. `req.sendBodyComplete(@constCast(json_body))`
3. `req.receiveHead(&redirect_buf)`
4. Check `response.head.status`
5. For body: `req.reader.bodyReader(&transfer_buf, response.head.transfer_encoding, response.head.content_length)`
6. `reader.allocRemaining(allocator, .unlimited)` for full body, or loop `reader.readVec()` for streaming

```zig
// typecast-zig/examples/tts.zig
const std = @import("std");
const typecast = @import("typecast");

pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    const allocator = gpa.allocator();

    var client = typecast.Client.init(allocator, .{
        .api_key = std.posix.getenv("TYPECAST_API_KEY") orelse @panic("set TYPECAST_API_KEY"),
    });
    defer client.deinit();

    // Regular TTS
    const response = try client.textToSpeech(.{
        .voice_id = "tc_672c5f5ce59fac2a48faeaee",
        .text = "Hello from the Zig SDK!",
        .model = .ssfm_v30,
    });
    defer allocator.free(response.audio_data);

    const file = try std.fs.cwd().createFile("output.wav", .{});
    defer file.close();
    try file.writeAll(response.audio_data);

    std.debug.print("Saved {d} bytes, duration: {d:.1}s\n", .{ response.audio_data.len, response.duration });
}
```
